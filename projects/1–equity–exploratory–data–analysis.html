from __visualization.colors import colors_fader, colormap

colors = colors_fader(color1="#0064d2", color2="#e53238", number=10)
cmap = colormap(colors)

# calculate the correlation matrix
correlation_matrix = returns.corr()

# create a heatmap of the correlations using the custom color map
plt.figure(figsize=(10, 10))
seaborn.heatmap(
    correlation_matrix,
    vmin=-1,
    vmax=1,
    annot=True,
    fmt=".2f",
    cmap=cmap,
    square=True,
    cbar_kws={
        "shrink": 0.8,
        "ticks": np.linspace(-1, 1, 10 + 1),
    },
)
